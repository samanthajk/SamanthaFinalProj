The purpose of this program is to preform a quiz about me (Sami), it checks for right and wrong answers. In this example of PHP code I demonstrate the use of arrays with a bit of UI work as well. 
  
< ?php
$SamiStuff = array(
     "Fave Color" => "Yellow",
     "Fave Food" => "Cake",
     "Fave Musical" => "Hamilton",
     "Hair Color" => "Brown",
     "Eye Color" => "Brown",
     "Dogs Name" => "Nikki",
	 "Born" => "Missouri"   
); //all info is set here
if (isset($_POST['submit'])) {
     $Answers = $_POST['answers'];
     if (is_array($Answers)) {
          foreach ($Answers as
               $Stuff => $Response) {
               $Response =
                    stripslashes($Response);
               if (strlen($Response)>0) {
                    if (strcasecmp(
                         $SamiStuff[$Stuff],
                         $Response)==0)
                         echo "
Correct! You got sami's $Stuff right, it was " .
                              $SamiStuff[$Stuff] .
                              ".
\n"; //correct output
                    else
                         echo "
Sorry, you got $Stuff wrong.
\n"; //incorrect output
               }
               else
                    echo "
You did not enter a
                         value for
                         $Stuff.
\n"; //invalid output
          }
     }
     echo "< p>< a href='SamiQuiz.php'>
          Try again?\n";
}
else {
     echo "
\n";
     foreach ($SamiStuff as
          $Stuff => $Response)
          echo "Do you know this about sami? $Stuff:
               
\n"; //setting each question format
          echo " ";
          echo "\n";
          echo "
\n";
}
?>
