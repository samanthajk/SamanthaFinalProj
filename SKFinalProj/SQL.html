This program is a small DB for a hospital, its employees, patients, and financial services. It displays the use of primary key, foreign key, and relationship mapping. 
  
-- ****************** SqlDBM: MySQL ******************;
-- ***************************************************;


-- ************************************** `financial`

CREATE TABLE `financial`
(
 `PayID`     integer NOT NULL ,
 `PayDate`   date NOT NULL ,
 `StartDate` datetime NOT NULL ,
 `Salary`    double NOT NULL ,
PRIMARY KEY (`PayID`)
);
-- ************************************** `employee`

CREATE TABLE `employee`
(
 `id`        integer NOT NULL ,
 `FName`     varchar(45) NOT NULL ,
 `LName`     varchar(45) NOT NULL ,
 `StartDate` datetime NOT NULL ,
 `PatientID` integer NOT NULL ,
 `PayID`     integer NOT NULL ,
PRIMARY KEY (`id`),
KEY `fkIdx_26` (`PatientID`),
CONSTRAINT `FK_26` FOREIGN KEY `fkIdx_26` (`PatientID`) REFERENCES `patient` (`PatientID`),
KEY `fkIdx_35` (`PayID`),
CONSTRAINT `FK_35` FOREIGN KEY `fkIdx_35` (`PayID`) REFERENCES `financial` (`PayID`)
);
-- ************************************** `patient`

CREATE TABLE `patient`
(
 `PatientID` integer NOT NULL ,
 `PFName`    varchar(45) NOT NULL ,
 `PLName`    varchar(45) NOT NULL ,
 `StartDate` date NOT NULL ,
PRIMARY KEY (`PatientID`)
);
-- ************************************** `location`

CREATE TABLE `location`
(
 `LCName`    varchar(45) NOT NULL ,
 `id`        integer NOT NULL ,
 `PayID`     integer NOT NULL ,
 `PatientID` integer NOT NULL ,
PRIMARY KEY (`LCName`),
KEY `fkIdx_42` (`id`),
CONSTRAINT `FK_42` FOREIGN KEY `fkIdx_42` (`id`) REFERENCES `employee` (`id`),
KEY `fkIdx_45` (`PayID`),
CONSTRAINT `FK_45` FOREIGN KEY `fkIdx_45` (`PayID`) REFERENCES `financial` (`PayID`),
KEY `fkIdx_48` (`PatientID`),
CONSTRAINT `FK_48` FOREIGN KEY `fkIdx_48` (`PatientID`) REFERENCES `patient` (`PatientID`)
);
