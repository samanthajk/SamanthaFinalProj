This program is one that allows the user to enter in variables and calculate the volume/surface area. In this example of C++ code I use simple user input and logical math to calculate the end product. 
  
#include < iostream>
#include < iomanip>
#include  < cmath>

using namespace std;

const double PI = 3.14159;

int main()
{
	double height = 0.0;
	double radius = 0.0;
	int hold;

	cout << fixed << showpoint << setprecision(2);

	cout << "Enter the height of the cylinder: ";
	cin >> height;
	cout << endl;

	cout << "Enter the radius of the cylinder: ";
	cin >> radius;
	cout << endl;

	cout << "Volume of the cylinder = "
		<< PI * (radius, 2.0) * height << endl;

	cout << "Surface area: "
		<< 2 * PI * radius * height + PI * 2 * pow(radius, 2.0)
		<< endl;

	cin >> hold;

	return 0;
}
