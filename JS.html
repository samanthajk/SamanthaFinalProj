The purpose of this program is to display what the user types in the input box. In this example of JS code, the main purpose was to solve a bug in the code, but it makes use of commonly used things (addEventListener, and strings).

main.js

  
'use strict';

populateInfo() {
	if (location.search) {
		var greeting = location.search;
		greeting = greeting.replace("+"," ");
		greeting = greeting.substring(greeting.lastIndexOf("=")
		+ 1);
		document.getElementById("greetingtext").innerHTML =
			decodeURIComponent(greeting);
	}
}

if (window.addEventListener) {
	window.addEventListener("load", populateInfo, false);
} else if (window.attachEvent) {
	window.attachEvent("onload", populateInfo);
}
